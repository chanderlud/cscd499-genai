1) Create a centered popup window

**Spec:** Write a Win32 application that creates a borderless (popup) window centered on the primary screen. The window must have a fixed size of 600x400 and a gray background (using the system color COLOR_WINDOWFRAME). The window should close when the system sends a WM_DESTROY message (e.g., via Alt+F4). The program must run a standard Win32 message loop until the window is closed.

**Constraints:**
- Use the `windows` crate (version 0.58.0 or later).
- The window class must be registered with the name "CenteredWindowClass".
- The window must be created with the style `WS_POPUP | WS_VISIBLE`.
- The background must be set to `COLOR_WINDOWFRAME` via `GetStockObject`.
- The window must be centered on the primary monitor using `GetSystemMetrics` and `SetWindowPos`.
- The program must handle the `WM_DESTROY` message by posting a quit message.

**Signature:**
```rust
fn main() -> windows::core::Result<()>
```

**Example:**
```rust
use windows::core::{w, Result};
use windows::Win32::UI::WindowsAndMessaging::{CreateWindowExW, WS_POPUP, WS_VISIBLE, CW_USEDEFAULT, GetSystemMetrics, SetWindowPos, SM_CXSCREEN, SM_CYSCREEN, HWND_TOP, SWP_NOSIZE, SWP_NOZORDER, SWP_SHOWWINDOW, GetMessageW, TranslateMessage, DispatchMessageW, MSG, RegisterClassW, WNDCLASSW, CS_HREDRAW, CS_VREDRAW, LoadCursorW, IDC_ARROW, GetStockObject, COLOR_WINDOWFRAME, GET_STOCK_OBJECT_FLAGS, HBRUSH, DefWindowProcW, PostQuitMessage, WM_DESTROY};
use windows::Win32::Foundation::{HINSTANCE, HWND, LPARAM, LRESULT, WPARAM};
use windows::Win32::System::LibraryLoader::GetModuleHandleW;

fn main() -> Result<()> {
    unsafe {
        let instance = GetModuleHandleW(None)?;
        let window_class = w!("CenteredWindowClass");

        let background = GetStockObject(GET_STOCK_OBJECT_FLAGS(COLOR_WINDOWFRAME.0));

        let wnd_class = WNDCLASSW {
            hInstance: instance.into(),
            lpszClassName: window_class,
            lpfnWndProc: Some(window_proc),
            style: CS_HREDRAW | CS_VREDRAW,
            hCursor: LoadCursorW(None, IDC_ARROW)?,
            hbrBackground: HBRUSH(background.0),
            ..Default::default()
        };

        RegisterClassW(&wnd_class);

        let hwnd = CreateWindowExW(
            Default::default(),
            window_class,
            w!("Centered Window"),
            WS_POPUP | WS_VISIBLE,
            CW_USEDEFAULT, CW_USEDEFAULT,
            600, 400,
            None, None, Some(HINSTANCE(instance.0)), None,
        )?;

        let screen_width = GetSystemMetrics(SM_CXSCREEN);
        let screen_height = GetSystemMetrics(SM_CYSCREEN);
        SetWindowPos(hwnd, HWND_TOP, (screen_width - 600) / 2, (screen_height - 400) / 2, 0, 0, SWP_NOSIZE | SWP_NOZORDER | SWP_SHOWWINDOW)?;

        let mut msg = MSG::default();
        while GetMessageW(&mut msg, None, 0, 0).into() {
            TranslateMessage(&msg);
            DispatchMessageW(&msg);
        }

        Ok(())
    }
}

extern "system" fn window_proc(hwnd: HWND, msg: u32, wparam: WPARAM, lparam: LPARAM) -> LRESULT {
    match msg {
        WM_DESTROY => {
            unsafe { PostQuitMessage(0) };
            LRESULT(0)
        }
        _ => unsafe { DefWindowProcW(hwnd, msg, wparam, lparam) },
    }
}
```
