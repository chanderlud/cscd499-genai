**Spec:**
- Create a layered overlay window that can be dragged by clicking and dragging anywhere within its bounds
- Implement window resizing by dragging from any edge or corner (8-directional resize handles)
- Maintain the window's topmost and click-through properties except during active drag/resize operations
- Provide functions to start/stop drag and resize operations based on mouse position
- Provide helper functions for coordinate transformations between screen and client space

**Constraints:**
- Use `WS_EX_LAYERED | WS_EX_TOPMOST | WS_EX_TOOLWINDOW | WS_EX_TRANSPARENT` extended styles for the window
- Window must be initially click-through (`WS_EX_TRANSPARENT`) but temporarily become interactive during drag/resize
- Resize handles should be 8 pixels wide along edges and 8×8 pixels at corners
- Return `windows::core::Result<T>` for fallible functions (with `T` being `HWND` for creation and `()` for others) and `Option<ResizeDirection>` for hit testing
- Must properly handle coordinate transformations between screen and client space

**Signature:**
```rust
/// Create a draggable, resizable overlay window
pub fn create_draggable_overlay(
    class_name: &str,
    initial_rect: &RECT,
) -> windows::core::Result<HWND>;

/// Test if a screen point hits a resize handle (returns direction)
pub fn hit_test_resize(hwnd: HWND, screen_point: &POINT) -> Option<ResizeDirection>;

/// Begin drag operation (call on WM_LBUTTONDOWN when not on resize handle)
pub fn begin_drag(hwnd: HWND, screen_point: &POINT) -> windows::core::Result<()>;

/// Begin resize operation (call on WM_LBUTTONDOWN when on resize handle)
pub fn begin_resize(hwnd: HWND, screen_point: &POINT, direction: ResizeDirection) -> windows::core::Result<()>;

/// Update drag/resize (call on WM_MOUSEMOVE)
pub fn update_drag_resize(hwnd: HWND, screen_point: &POINT) -> windows::core::Result<()>;

/// End drag/resize operation (call on WM_LBUTTONUP)
pub fn end_drag_resize(hwnd: HWND) -> windows::core::Result<()>;

/// Convert client coordinates to screen coordinates
pub fn client_to_screen(hwnd: HWND, point: &POINT) -> POINT;

/// Convert screen coordinates to client coordinates
pub fn screen_to_client(hwnd: HWND, point: &POINT) -> POINT;

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum ResizeDirection {
    North,
    South,
    East,
    West,
    NorthEast,
    NorthWest,
    SouthEast,
    SouthWest,
}
```

**Example:**
```rust
let hwnd = create_draggable_overlay("MyOverlay", &RECT { left: 100, top: 100, right: 400, bottom: 300 }).unwrap();

// In window procedure:
match msg {
    WM_LBUTTONDOWN => {
        let point = POINT { x: (lparam.0 & 0xFFFF) as i16 as i32, y: (lparam.0 >> 16) as i16 as i32 };
        let screen_point = client_to_screen(hwnd, &point);
        
        if let Some(direction) = hit_test_resize(hwnd, &screen_point) {
            begin_resize(hwnd, &screen_point, direction).unwrap();
        } else {
            begin_drag(hwnd, &screen_point).unwrap();
        }
    }
    WM_MOUSEMOVE => {
        let point = POINT { x: (lparam.0 & 0xFFFF) as i16 as i32, y: (lparam.0 >> 16) as i16 as i32 };
        let screen_point = client_to_screen(hwnd, &point);
        update_drag_resize(hwnd, &screen_point).unwrap();
    }
    WM_LBUTTONUP => {
        end_drag_resize(hwnd).unwrap();
    }
    _ => DefWindowProcW(hwnd, msg, wparam, lparam),
}
```
